async function handleRegister() { const user = document.getElementById('username').value; const pass = document.getElementById('password').value; const email = document.getElementById('email').value; if (!user || !pass || !email) { return alert("กรุณากรอกข้อมูลให้ครบถ้วน"); } try { // ตรวจสอบว่า URL คือ http://localhost:3000 ไม่ใช่พอร์ตอื่น const response = await fetch('http://localhost:3000/api/register', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: user, password: pass, email: email }) }); const data = await response.json(); if (response.ok) { alert("สมัครสมาชิกสำเร็จ!"); window.location.href = 'login.html'; // สมัครเสร็จแล้วส่งไปหน้า Login } else { // ถ้าชื่อซ้ำ หรือ Supabase มีปัญหา จะโชว์ Error จาก Server alert(data.message || "สมัครสมาชิกไม่สำเร็จ"); } } catch (error) { console.error("Register Error:", error); alert("ติดต่อ Server ไม่ได้! ตรวจสอบว่าใน Terminal ขึ้นคำว่า 'Server พร้อมทำงาน' หรือยัง?"); } }
